```typescript
interface ScoringFactors {
  guaranteedDemand: number; // 0-100
  automationLevel: number;
  tokenEfficiency: number;
  profitMargin: number;
  maintenanceCost: number;
  legalRisk: number;
  competitionSaturation: number;
  ethicalRisk: number;
}

const WEIGHTS: Record<keyof ScoringFactors, number> = {
  guaranteedDemand: 0.2,
  automationLevel: 0.15,
  tokenEfficiency: 0.15,
  profitMargin: 0.2,
  maintenanceCost: -0.1, // Negative weight
  legalRisk: -0.1,
  competitionSaturation: -0.1,
  ethicalRisk: -0.1,
};

function calculateCompositeScore(factors: ScoringFactors): number {
  let total = 0;
  let weightSum = 0;
  for (const [key, value] of Object.entries(factors)) {
    const weight = WEIGHTS[key as keyof ScoringFactors];
    const normalized = weight < 0 ? (100 - value) : value; // Invert for negative factors
    total += normalized * Math.abs(weight);
    weightSum += Math.abs(weight);
  }
  return Math.round(total / weightSum);
}

// Usage in scoreOpportunity
async function scoreOpportunity(opportunity: Partial<BusinessOpportunity>): Promise<CompositeScores> {
  // Fetch factors via AI research or static rules
  const factors = await researchFactors(opportunity); // Implement with modelRouter
  return { ...factors, composite: calculateCompositeScore(factors) };
}
```

### API Router Design (Model Router)

Extend `modelRouter.ts` for task-based routing. Design: Priority queue by cost/capability, with fallbacks.

#### Design Principles
- **Routing Logic**: Sort models by cost ascending for task; fallback on error.
- **Edge Cases**: No models available -> throw AGENT_NO_API_CONFIGURED.
- **Implications**: Logs for auditing; integrate with token usage.

#### Code
```typescript
const MODEL_CONFIGS: ModelConfig[] = [
  { provider: 'perplexity', model: 'pplx-70b', costPer1kTokens: 0.0005, capabilities: ['research', 'analysis'] },
  // Add Grok, GPT, etc.
];

function selectModel(taskType: TaskType, userConfigs: ApiConfig[]): ModelConfig {
  const available = MODEL_CONFIGS.filter(m => m.capabilities.includes(taskType))
    .sort((a, b) => a.costPer1kTokens - b.costPer1kTokens);
  if (!available.length) throw new Error('No model for task');
  return available[0]; // Cheapest first
}

async function executeWithFallback<T>(taskType: TaskType, prompt: string, userConfigs: ApiConfig[]): Promise<T> {
  const models = MODEL_CONFIGS.filter(m => m.capabilities.includes(taskType)).sort(/* by cost */);
  for (const model of models) {
    try {
      // Call API based on provider, e.g., openai.chat.completions.create({ model: model.model, prompt })
      const response = await callModelApi(model, prompt);
      logTokenUsage(/* details */);
      return response as T;
    } catch (e) {
      console.warn(`Model ${model.model} failed: ${e}`);
    }
  }
  throw new Error('All models failed');
}
```
